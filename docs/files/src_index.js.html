	function jqScope(el, parentSelector, namespace) {

		var localScope;

		// the local data
		var localData = el.prefixedData(namespace);

		// get the parent
		var parentEl = el.parent().closest(parentSelector);

		if (parentEl.length > 0) {
			// there is a parent

			var parentScope = jqScope(parentEl, parentSelector, namespace);

			localScope = parentScope.create(localData);

		} else {
			// this is root

			var scopeId = namespace ? 'scope-' + namespace : 'scope';

			// check if there is a 'scope-$namespace'
			localScope = el.data(scopeId);

			if (!localScope) {
				// the local scope has not been created yet

				// create
				localscope = scope(localData);

				// and store
				el.data(scopeId, localScope);
			}

		}

		// return.
		return localScope;
	}



	$.prototype.scope = function jqScope(parentSelector, namespace, data) {

	};

	exports.get = function getScope(el, namespace) {
		return el.data(buildScopeName(namespace));
	}

	exports.set = function setScopeData(el /* namespace, data/key, value */) {

		var scopeName, data;

		if (arguments.length === 2) {
			// arguments = [el, data]
			scopeName = buildScopeName();
			data = arguments[1];

		} else if (arguments.length === 3) {
			// arguments = [el, namespace, data]
			scopeName = buildScopeName(arguments[1]);
			data = arguments[2];

		} else if (arguments.length === 4) {
			// arguments = [el, namespace, key, value]
			scopeName = buildScopeName(arguments[1]);
			data = ({})[arguments[2]] = arguments[3];
		}

		// get scope
		var scope = el.data(scopeName);

		// set data
		_.extend(scope, data);

		return el;
	}


	exports.inherit = function inheritScope(el, parent, namespace) {

		var scopeName = buildScopeName(namespace),
			// get the scope from the parent
			parentScope = exports.get(parent, namespace);

		el.data(scopeName, _.create(parentScope));

		return el;
	};


	exports.create = function createScope(el /* parent, namespace, data */) {
		var parent, scopeName, data;

		if (arguments.length === 2) {
			// arguments = [el, data];
			scopeName = buildScopeName();
		}

		el.data(buildScopeName(namespace), data);

		return data;
	}
